class Main { // class Main where the program execution starts

    static class Node { // class Node to represent each node of the tree
        int data; // stores the value of the node
        Node left, right; // references/pointers to left and right child nodes

        Node(int data) { // constructor to initiate the node with a value
            this.data = data; // assign the given value to the node
            this.left = null; // left child is null, initially
            this.right = null; // right child is null, initially
        }
    }

    public static void nodeDistanceK(Node root, int k) { 
        // function to print all nodes at distance k from the root

        if (root == null) { 
            // if the current node is null, recursion stop
            return;
        }

        if (k == 0) { 
            // when distance becomes 0, print the current node's value
            System.out.print(root.data + " ");
            return; // stop further recursive calls
        }

        // recursion go to left and right branch, reducing k by 1
        nodeDistanceK(root.left, k - 1);
        nodeDistanceK(root.right, k - 1);
    }

    public static void main(String[] args) { 
        // main method: program execution starts here

        // creating a sample binary tree
        Node root = new Node(1); 
        root.left = new Node(2);
        root.right = new Node(3);
        root.left.left = new Node(4);
        root.left.right = new Node(5);

        int k = 2; // distance from root
        nodeDistanceK(root, k); // function call
    }
}
